## 2. 架构概述

### 2.1 设计原则

1. **模块化设计**：各功能模块低耦合、高内聚，便于独立开发和维护
2. **跨平台兼容**：支持 Windows、macOS、Linux 三大平台
3. **可扩展性**：技能系统支持热加载，插件架构支持第三方扩展
4. **安全性**：沙箱隔离、权限控制、数据加密等多重安全保障
5. **性能优先**：异步执行、资源控制、缓存优化
6. **用户友好**：简洁直观的界面，清晰的操作流程

### 2.2 架构风格

采用**分层架构** + **插件架构** + **事件驱动**的混合架构风格：

- **分层架构**：UI层、对话引擎层、业务逻辑层、技能引擎层、平台适配层、数据层
- **插件架构**：技能和插件作为独立模块动态加载
- **事件驱动**：基于发布-订阅模式的消息传递机制

### 2.3 技术选型

| 组件 | 技术方案 | 理由 |
|------|---------|------|
| 桌面框架 | Tauri 1.x | 轻量、安全、性能好、包体小 |
| 前端框架 | React 18 + TypeScript | 生态成熟、组件化开发、类型安全 |
| UI 组件库 | Ant Design | 企业级组件、开箱即用、主题定制 |
| 状态管理 | Zustand | 轻量、API 简洁、性能优秀 |
| 后端语言 | Rust | 安全、高性能、系统操作支持好 |
| 数据库 | SQLite 3 | 嵌入式、零配置、跨平台 |
| 配置格式 | YAML | 人类可读、注释友好、层次清晰 |
| 日志系统 | tracing | 结构化日志、性能优秀、上下文支持 |
| 网络服务 | hyper + tokio | 异步、高性能、生态完善 |
| 序列化 | serde | 类型安全、零成本抽象 |
| 前端构建 | Vite | 快速、现代、ESM 原生 |

```rust
// src/core/app.rs
pub struct Application {
    config: Arc<RwLock<Config>>,
    event_bus: Arc<EventBus>,
    skill_engine: Arc<SkillEngine>,
    dialog_engine: Arc<DialogEngine>,
    services: Arc<ServiceContainer>,
    database: Arc<Database>,
}

impl Application {
    /// 初始化应用
    pub async fn initialize(config_path: &Path) -> Result<Self, AppError> {
        // 1. 加载配置
        let config = Config::load(config_path)?;

        // 2. 初始化数据库
        let database = Database::new(&config.database_path).await?;

        // 3. 初始化事件总线
        let event_bus = Arc::new(EventBus::new());

        // 4. 初始化服务容器
        let services = Arc::new(ServiceContainer::new(
            config.clone(),
            database.clone(),
            event_bus.clone(),
        )?);

        // 5. 初始化技能引擎
        let skill_engine = Arc::new(SkillEngine::new(
            config.skill_dir.clone(),
            services.clone(),
            event_bus.clone(),
        )?);

        // 6. 初始化对话引擎
        let dialog_engine = Arc::new(DialogEngine::new(
            skill_engine.clone(),
            services.clone(),
            event_bus.clone(),
        )?);

        Ok(Self {
            config: Arc::new(RwLock::new(config)),
            event_bus,
            skill_engine,
            dialog_engine,
            services,
            database,
        })
    }

    /// 启动应用
    pub async fn start(&self) -> Result<(), AppError> {
        // 1. 启动 Web 服务器
        self.start_web_server().await?;

        // 2. 加载技能
        self.skill_engine.load_all_skills().await?;

        // 3. 启动后台任务
        self.start_background_tasks().await?;

        Ok(())
    }

    /// 停止应用
    pub async fn shutdown(&self) -> Result<(), AppError> {
        // 1. 停止 Web 服务器
        self.stop_web_server().await?;

        // 2. 卸载技能
        self.skill_engine.unload_all_skills().await?;

        // 3. 保存状态
        self.save_state().await?;

        Ok(())
    }
}
```

```rust
use tokio::sync::broadcast;
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum AppEvent {
    // 技能事件
    SkillLoaded { skill_id: String },
    SkillUnloaded { skill_id: String },
    SkillExecuted { skill_id: String, result: ExecutionResult },
    SkillFailed { skill_id: String, error: String },

    // 对话事件
    MessageReceived { user_id: String, content: String },
    IntentRecognized { intent: String, confidence: f32 },
    DialogStarted { session_id: String },
    DialogEnded { session_id: String },

    // 系统事件
    ConfigChanged,
    ServiceStarted { service_name: String },
    ServiceStopped { service_name: String },
    ErrorOccurred { source: String, error: String },
}

pub struct EventBus {
    sender: broadcast::Sender<AppEvent>,
}

impl EventBus {
    pub fn new() -> Self {
        let (sender, _) = broadcast::channel(1000);
        Self { sender }
    }

    pub fn publish(&self, event: AppEvent) {
        let _ = self.sender.send(event);
    }

    pub fn subscribe(&self) -> broadcast::Receiver<AppEvent> {
        self.sender.subscribe()
    }
}
```

```rust
use std::path::PathBuf;
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SkillManifest {
    pub name: String,
    pub version: String,
    pub description: String,
    pub author: Option<String>,
    pub tags: Vec<String>,
    pub main: String,
    pub permissions: Permissions,
    pub dependencies: Vec<Dependency>,
    pub dialog: Option<DialogConfig>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Permissions {
    pub file: Vec<FilePermission>,
    pub network: Vec<NetworkPermission>,
    pub system: Vec<SystemPermission>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct FilePermission {
    pub path: String,
    pub access: String, // "read", "write", "readwrite"
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct DialogConfig {
    pub intents: Vec<IntentConfig>,
    pub slots: Vec<SlotConfig>,
    pub response: ResponseConfig,
}

pub struct SkillLoader {
    skill_dir: PathBuf,
}

impl SkillLoader {
    pub fn new(skill_dir: PathBuf) -> Self {
        Self { skill_dir }
    }

    /// 加载技能
    pub async fn load_skill(&self, skill_path: &Path) -> Result<Skill, SkillError> {
        // 1. 读取技能清单
        let manifest_path = skill_path.join("skill.yaml");
        let manifest = Self::load_manifest(&manifest_path).await?;

        // 2. 验证权限
        self.validate_permissions(&manifest)?;

        // 3. 验证依赖
        self.validate_dependencies(&manifest)?;

        // 4. 加载技能代码
        let skill_code = self.load_skill_code(skill_path, &manifest).await?;

        // 5. 创建技能实例
        let skill = Skill::new(manifest, skill_code);

        Ok(skill)
    }

    /// 批量加载技能
    pub async fn load_all(&self) -> Result<Vec<Skill>, SkillError> {
        let mut skills = Vec::new();

        for entry in fs::read_dir(&self.skill_dir)? {
            let entry = entry?;
            if entry.path().is_dir() {
                if let Ok(skill) = self.load_skill(&entry.path()).await {
                    skills.push(skill);
                }
            }
        }

        Ok(skills)
    }
}
```

```rust
// src/skill/executor.rs
use std::process::Command;

pub struct SkillExecutor {
    sandbox: Arc<Sandbox>,
    services: Arc<ServiceContainer>,
}

impl SkillExecutor {
    pub async fn execute(
        &self,
        skill: &Skill,
        params: serde_json::Value,
    ) -> Result<ExecutionResult, ExecutionError> {
        // 1. 检查权限
        self.check_permissions(skill).await?;

        // 2. 准备执行环境
        let env = self.prepare_env(skill, &params).await?;

        // 3. 在沙箱中执行
        let result = self.sandbox.execute(skill, env).await?;

        // 4. 处理结果
        let processed = self.process_result(result).await?;

        Ok(processed)
    }

    /// 异步执行
    pub async fn execute_async(
        &self,
        skill: &Skill,
        params: serde_json::Value,
    ) -> String {
        let task_id = uuid::Uuid::new_v4().to_string();

        let skill = skill.clone();
        let sandbox = self.sandbox.clone();

        tokio::spawn(async move {
            let result = sandbox.execute(&skill, params).await;
            // 通知任务完成
            event_bus.publish(AppEvent::SkillExecuted {
                skill_id: skill.id.clone(),
                result,
            });
        });

        task_id
    }
}
```

```rust
// src/skill/sandbox.rs
#[cfg(target_os = "linux")]
pub struct Sandbox {
    cgroup: Cgroup,
    seccomp: SeccompFilter,
}

#[cfg(target_os = "windows")]
pub struct Sandbox {
    job_object: JobObject,
    app_container: Option<AppContainer>,
}

#[cfg(target_os = "macos")]
pub struct Sandbox {
    sandbox_profile: String,
}

impl Sandbox {
    pub async fn execute(
        &self,
        skill: &Skill,
        env: ExecutionEnv,
    ) -> Result<ExecutionResult, SandboxError> {
        // 根据平台实现不同的沙箱执行
        #[cfg(target_os = "linux")]
        {
            self.execute_linux(skill, env).await
        }

        #[cfg(target_os = "windows")]
        {
            self.execute_windows(skill, env).await
        }

        #[cfg(target_os = "macos")]
        {
            self.execute_macos(skill, env).await
        }
    }

    #[cfg(target_os = "linux")]
    async fn execute_linux(
        &self,
        skill: &Skill,
        env: ExecutionEnv,
    ) -> Result<ExecutionResult, SandboxError> {
        // 使用 cgroups 和 seccomp 隔离
        // 1. 创建 cgroup
        let cgroup_path = self.cgroup.create_child()?;

        // 2. 设置资源限制
        self.cgroup.set_cpu_limit(cgroup_path, 50)?; // 50% CPU
        self.cgroup.set_memory_limit(cgroup_path, 1024 * 1024 * 1024)?; // 1GB 内存

        // 3. 应用 seccomp 过滤
        self.seccomp.apply(skill.id())?;

        // 4. 执行技能
        let result = tokio::process::Command::new(&skill.executable)
            .args(&skill.args)
            .envs(env.variables)
            .current_dir(env.working_dir)
            .output()
            .await?;

        // 5. 清理
        self.cgroup.remove_child(cgroup_path)?;

        Ok(ExecutionResult::from_output(result))
    }
}
```

```rust
// src/dialog/intent.rs
use std::collections::HashMap;

#[derive(Debug, Clone)]
pub struct Intent {
    pub name: String,
    pub confidence: f32,
    pub skill_id: String,
    pub slots: HashMap<String, String>,
}

pub struct IntentRecognizer {
    keyword_matcher: KeywordMatcher,
    regex_matcher: RegexMatcher,
    ml_classifier: Option<MLClassifier>,
    llm_client: Option<LLMClient>,
}

impl IntentRecognizer {
    pub async fn recognize(&self, text: &str) -> Option<Intent> {
        // 1. 关键词匹配 (准确率 > 95%)
        if let Some(intent) = self.keyword_matcher.match(text).await {
            if intent.confidence > 0.95 {
                return Some(intent);
            }
        }

        // 2. 正则表达式 (准确率 > 90%)
        if let Some(intent) = self.regex_matcher.match(text).await {
            if intent.confidence > 0.90 {
                return Some(intent);
            }
        }

        // 3. ML 分类 (准确率 > 85%)
        if let Some(classifier) = &self.ml_classifier {
            if let Some(intent) = classifier.predict(text).await {
                if intent.confidence > 0.85 {
                    return Some(intent);
                }
            }
        }

        // 4. LLM 语义理解
        if let Some(llm) = &self.llm_client {
            return llm.understand(text).await;
        }

        None
    }
}
```

```rust
// src/dialog/session.rs
use std::collections::HashMap;
use std::time::{Duration, Instant};

#[derive(Debug, Clone)]
pub struct Session {
    pub id: String,
    pub user_id: String,
    pub chat_id: String,
    pub platform: String,
    pub current_intent: Option<String>,
    pub slots: HashMap<String, String>,
    pub messages: Vec<Message>,
    pub created_at: Instant,
    pub last_active: Instant,
}

pub struct SessionManager {
    sessions: Arc<RwLock<HashMap<String, Session>>>,
    timeout: Duration,
}

impl SessionManager {
    pub fn new(timeout: Duration) -> Self {
        Self {
            sessions: Arc::new(RwLock::new(HashMap::new())),
            timeout,
        }
    }

    /// 获取或创建会话
    pub async fn get_or_create(
        &self,
        user_id: &str,
        chat_id: &str,
        platform: &str,
    ) -> Session {
        let session_id = format!("{}:{}:{}", platform, user_id, chat_id);
        let mut sessions = self.sessions.write().await;

        if let Some(session) = sessions.get_mut(&session_id) {
            // 检查是否超时
            if session.last_active.elapsed() > self.timeout {
                // 创建新会话
                let new_session = self.create_session(user_id, chat_id, platform);
                sessions.insert(session_id.clone(), new_session.clone());
                new_session
            } else {
                // 更新活跃时间
                session.last_active = Instant::now();
                session.clone()
            }
        } else {
            // 创建新会话
            let new_session = self.create_session(user_id, chat_id, platform);
            sessions.insert(session_id.clone(), new_session.clone());
            new_session
        }
    }

    /// 更新会话
    pub async fn update(&self, session: Session) {
        let mut sessions = self.sessions.write().await;
        sessions.insert(session.id.clone(), session);
    }

    /// 添加消息
    pub async fn add_message(&self, session_id: &str, message: Message) {
        let mut sessions = self.sessions.write().await;
        if let Some(session) = sessions.get_mut(session_id) {
            session.messages.push(message);
            session.last_active = Instant::now();
        }
    }

    /// 清理过期会话
    pub async fn cleanup_expired(&self) {
```

```rust
                path: entry.path(),
                is_dir: metadata.is_dir(),
                size: metadata.len(),
                modified: metadata.modified()?.into(),
            });
```

```rust
use hyper::{Client, Request, Body, Method, header};
use hyper_tls::HttpsConnector;

pub struct NetworkService {
    client: Client<HttpsConnector<hyper::client::HttpConnector>>,
}

impl NetworkService {
    pub fn new() -> Self {
        let https = HttpsConnector::new();
        let client = Client::builder()
            .pool_max_idle_per_host(10)
            .build::<_, Body>(https);

        Self { client }
    }

    /// 发送 HTTP 请求
    pub async fn request(
        &self,
        method: Method,
        url: &str,
        headers: HashMap<String, String>,
        body: Option<String>,
    ) -> Result<HttpResponse, NetworkError> {
        let mut req_builder = Request::builder()
            .method(method)
            .uri(url);

        // 添加请求头
        for (key, value) in headers {
            req_builder = req_builder.header(key, value);
        }

        // 添加请求体
        let req = if let Some(body) = body {
            req_builder.body(Body::from(body))?
        } else {
            req_builder.body(Body::empty())?
        };

        // 发送请求
        let response = self.client.request(req).await?;

        // 解析响应
        let status = response.status();
        let headers = response
            .headers()
            .iter()
            .map(|(k, v)| (k.as_str().to_string(), v.to_str().unwrap().to_string()))
            .collect();

        let body_bytes = hyper::body::to_bytes(response.into_body()).await?;
        let body = String::from_utf8(body_bytes.to_vec())?;

        Ok(HttpResponse {
            status: status.as_u16(),
            headers,
            body,
        })
    }
}
```

```rust
            .uri(&url)
            .header("Authorization", format!("Bearer {}", access_token))

            chat_id: event.chat_id,

                None => String::new(),
```
