## 2. 安全架构概述

### 2.1 安全目标

1. **机密性 (Confidentiality)**：保护敏感数据不被未授权访问
2. **完整性 (Integrity)**：确保数据不被篡改
3. **可用性 (Availability)**：保证系统持续可用
4. **可追溯性 (Accountability)**：记录所有操作以便审计

```rust
        Self::verify_platform_signature(event)?;

        // 映射用户
        let user_id = Self::map_user_id(
            &event.platform,
            &event.user_id
        )?;

        // 加载或创建用户
        let user = Self::load_or_create_user(user_id, event)?;

        Ok(user)
    }

    /// 映射平台用户 ID 到系统用户 ID
    fn map_user_id(platform: &str, platform_user_id: &str) -> Result<String, AuthError> {
        // 使用 HMAC 生成用户 ID，防止平台 ID 泄露
        let key = Self::get_hmac_key()?;
        let data = format!("{}:{}", platform, platform_user_id);

        let user_id = hmac_sha256(&key, data.as_bytes());

        Ok(format!("user_{}", &user_id[..16]))
    }
}
```

```rust
            return Err(AuthError::InvalidKeyFormat);
        }

    pub async fn generate_api_key(&self, user_id: &str) -> String {
        let timestamp = chrono::Utc::now().timestamp();
        let random: String = (0..32)
            .map(|_| {
                let chars = b"abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789";

            })

        let key = format!("da_{}_{}_{}", timestamp, user_id, random);

        // 存储到数据库
        self.store_api_key(user_id, &key).await;

        key
    }
```

```rust
// src/auth/permission.rs

#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
pub struct Permission {
    pub resource: String,   // 资源类型：file/system/network
    pub action: String,     // 操作类型：read/write/execute
    pub scope: Option<String>, // 作用域：路径/域名等
}

impl Permission {
    /// 从字符串解析权限
    pub fn from_str(s: &str) -> Result<Self, PermissionError> {
        let parts: Vec<&str> = s.split(':').collect();

        if parts.len() < 2 {
            return Err(PermissionError::InvalidFormat);
        }

        Ok(Self {
            resource: parts[0].to_string(),
            action: parts[1].to_string(),
            scope: parts.get(2).map(|s| s.to_string()),
        })
    }

    /// 检查权限是否匹配
    pub fn matches(&self, other: &Permission) -> bool {
        if self.resource == "*" && self.action == "*" {
            return true;
        }

        if self.resource != other.resource && self.resource != "*" {
            return false;
        }

        if self.action != other.action && self.action != "*" {
            return false;
        }

        if let (Some(scope), Some(other_scope)) = (&self.scope, &other.scope) {
            return self.match_scope(scope, other_scope);
        }

        true
    }

    fn match_scope(&self, pattern: &str, target: &str) -> bool {
        // 支持通配符匹配
        if pattern == "*" {
            return true;
        }

        if pattern.ends_with("/*") {
            let prefix = &pattern[..pattern.len()-1];
            return target.starts_with(prefix);
        }

        pattern == target
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Role {
    pub id: String,
    pub name: String,
    pub permissions: Vec<Permission>,
    pub is_default: bool,
}
```

#### 3.2.2 权限检查

```rust
// src/auth/authorizer.rs

pub struct Authorizer {
    db: Arc<Database>,
    cache: Arc<Mutex<LruCache<String, Vec<Permission>>>>,
}

impl Authorizer {
    /// 检查权限
    pub async fn check_permission(
        &self,
        user_id: &str,
        permission: &Permission,
    ) -> Result<bool, AuthError> {
        // 1. 获取用户权限
        let user_permissions = self.get_user_permissions(user_id).await?;

        // 2. 检查是否有匹配的权限
        let has_permission = user_permissions
            .iter()
            .any(|p| p.matches(permission));

        // 3. 记录权限检查日志
        self.log_permission_check(user_id, permission, has_permission).await;

        Ok(has_permission)
    }

    /// 获取用户权限
    async fn get_user_permissions(&self, user_id: &str) -> Result<Vec<Permission>, AuthError> {
        // 检查缓存
        if let Some(permissions) = self.cache.lock().await.get(user_id) {
            return Ok(permissions.clone());
        }

        // 从数据库加载
        let user = self.db.get_user(user_id).await?
            .ok_or(AuthError::UserNotFound)?;

        let role = self.db.get_role(&user.role).await?
            .ok_or(AuthError::RoleNotFound)?;

        let permissions = role.permissions;

        // 缓存
        self.cache.lock().await.put(user_id.to_string(), permissions.clone());

        Ok(permissions)
    }

    /// 批量检查权限
    pub async fn check_permissions(
        &self,
        user_id: &str,
        permissions: &[Permission],
    ) -> Result<Vec<bool>, AuthError> {
        let results = futures::future::join_all(
            permissions.iter().map(|p| self.check_permission(user_id, p))
        ).await;

        Ok(results)
    }
}
```

```rust
// src/skill/sandbox/linux.rs

use std::process::Command;
use nix::sys::sched::{clone, CloneFlags};
use nix::sys::wait::{waitpid, WaitStatus};
use nix::unistd::Pid;

pub struct LinuxSandbox {
    cgroup_path: String,
    seccomp_filter: SeccompFilter,
}

impl LinuxSandbox {
    pub fn new(skill_id: &str) -> Result<Self, SandboxError> {
        let cgroup_path = format!("/sys/fs/cgroup/desktop-agent/{}", skill_id);

        // 创建 cgroup
        Self::create_cgroup(&cgroup_path)?;

        // 配置 seccomp
        let seccomp_filter = Self::create_seccomp_filter()?;

        Ok(Self {
            cgroup_path,
            seccomp_filter,
        })
    }

    fn create_cgroup(path: &str) -> Result<(), SandboxError> {
        fs::create_dir_all(path)?;

        // CPU 限制
        fs::write(format!("{}/cpu.max"), path, "50000 100000")?; // 50%

        // 内存限制
        fs::write(format!("{}/memory.max"), path, "1073741824")?; // 1GB

        // 进程数限制
        fs::write(format!("{}/pids.max"), path, "10")?;

        Ok(())
    }

    fn create_seccomp_filter() -> Result<SeccompFilter, SandboxError> {
        let mut filter = SeccompFilter::new(SeccompAction::Allow)?;

        // 禁用危险系统调用
        filter.add_rule(SeccompCondition::new(SCMP_SYS(execve))?.build(), SeccompAction::Errno(EPERM))?;
        filter.add_rule(SeccompCondition::new(SCMP_SYS(fork))?.build(), SeccompAction::Errno(EPERM))?;
        filter.add_rule(SeccompCondition::new(SCMP_SYS(chmod))?.build(), SeccompAction::Errno(EPERM))?;
        filter.add_rule(SeccompCondition::new(SCMP_SYS(chown))?.build(), SeccompAction::Errno(EPERM))?;

        Ok(filter)
    }

    pub async fn execute(&self, command: &mut Command) -> Result<ExecutionResult, SandboxError> {
        // 1. 设置 cgroup
        command.env("CGROUP_PATH", &self.cgroup_path);

        // 2. 应用 seccomp
        self.seccomp_filter.apply()?;

        // 3. 执行
        let output = command.output()?;

        // 4. 清理 cgroup
        self.cleanup_cgroup()?;

        Ok(ExecutionResult::from_output(output))
    }

    fn cleanup_cgroup(&self) -> Result<(), SandboxError> {
        // 删除 cgroup
        fs::remove_dir_all(&self.cgroup_path)?;
        Ok(())
    }
}
```

```rust
// src/skill/sandbox/windows.rs

use windows::Win32::System::JobObjects::*;

pub struct WindowsSandbox {
    job_object: HANDLE,
    app_container: Option<HANDLE>,
}

impl WindowsSandbox {
    pub fn new() -> Result<Self, SandboxError> {
        // 创建 Job Object
        unsafe {
            let job_object = CreateJobObjectW(None, None)?;

            // 设置基本限制
            let mut info = JOBOBJECT_BASIC_LIMIT_INFORMATION {
                PerJobUserTimeLimit: 0,
                PerProcessUserTimeLimit: 0,
                LimitFlags: JOB_OBJECT_LIMIT_KILL_ON_JOB_CLOSE |
                              JOB_OBJECT_LIMIT_DIE_ON_UNHANDLED_EXCEPTION,
                MinimumWorkingSetSize: 0,
                MaximumWorkingSetSize: 0,
                ActiveProcessLimit: 5,
                Affinity: 0,
                PriorityClass: 0,
                SchedulingClass: 0,
            };

            SetInformationJobObject(
                job_object,
                JobObjectBasicLimitInformation,
                &info as *const _ as *const _,
                std::mem::size_of::<JOBOBJECT_BASIC_LIMIT_INFORMATION>() as u32,
            )?;

            Ok(Self {
                job_object,
                app_container: None,
            })
        }
    }

    pub async fn execute(&self, command: &mut Command) -> Result<ExecutionResult, SandboxError> {
        unsafe {
            let mut si = STARTUPINFOW::default();
            let mut pi = PROCESS_INFORMATION::default();

            let cmd = command.get_program().to_str().unwrap();
            let args = command.get_args()
                .map(|s| s.to_str().unwrap())
                .collect::<Vec<_>>()
                .join(" ");

            let mut cmd_line = format!("{} {}", cmd, args);

            CreateProcessW(
                None,
                PWSTR(cmd_line.as_mut_ptr()),
                None,
                None,
                FALSE,
                0,
                None,
                None,
                &si,
                &mut pi,
            )?;

            // 将进程加入 Job Object
            AssignProcessToJobObject(self.job_object, pi.hProcess)?;

            // 等待进程结束
            WaitForSingleObject(pi.hProcess, INFINITE);

            // 获取退出码
            let mut exit_code = 0u32;
            GetExitCodeProcess(pi.hProcess, &mut exit_code);

            CloseHandle(pi.hProcess);
            CloseHandle(pi.hThread);

            Ok(ExecutionResult {
                success: exit_code == 0,
                exit_code: exit_code as i32,
                stdout: String::new(),
                stderr: String::new(),
            })
        }
    }
}
```

```rust
pub struct SkillAuditor;

    fn audit_python(code: &str, report: &mut AuditReport) -> Result<(), AuditError> {
        // 检查危险函数调用
        let dangerous_patterns = vec![
            (Regex::new(r"eval\s*\(").unwrap(), "使用了 eval() 函数，存在代码注入风险"),
            (Regex::new(r"exec\s*\(").unwrap(), "使用了 exec() 函数，存在代码注入风险"),
            (Regex::new(r"subprocess\.(call|run|Popen)\s*\(").unwrap(), "使用了 subprocess，可能执行系统命令"),
            (Regex::new(r"os\.system\s*\(").unwrap(), "使用了 os.system()，可能执行系统命令"),
            (Regex::new(r"import\s+pickle").unwrap(), "使用了 pickle，可能存在反序列化漏洞"),
            (Regex::new(r"open\s*\(\s*[\"']\s*/").unwrap(), "尝试访问系统根目录"),
        ];

        for (pattern, message) in dangerous_patterns {
            if pattern.is_match(code) {
                report.add_warning(message.to_string());
            }
        }

        // 检查硬编码密钥
        if Regex::new(r"(api_key|secret|password|token)\s*=\s*[\"']([a-zA-Z0-9]{20,})[\"']").unwrap().is_match(code) {
            report.add_error("检测到硬编码的敏感信息".to_string());
        }

        // 检查网络请求
        if Regex::new(r"(requests|urllib|httpx)\.").unwrap().is_match(code) {
            report.add_info("使用了网络请求，请确保已声明网络权限".to_string());
        }

        Ok(())
    }

    fn audit_javascript(code: &str, report: &mut AuditReport) -> Result<(), AuditError> {
        // JavaScript 特定的检查

            (Regex::new(r"eval\s*\(").unwrap(), "使用了 eval() 函数，存在代码注入风险"),
```

### 5.1 Webhook 签名验证

```rust
// src/security/webhook.rs

use hmac::{Hmac, Mac};
use sha2::Sha256;

type HmacSha256 = Hmac<Sha256>;

pub struct WebhookVerifier {
    secret: String,
}

impl WebhookVerifier {
    pub fn new(secret: String) -> Self {
        Self { secret }
    }

    /// 验证飞书签名
    pub fn verify_feishu(
        &self,
        timestamp: &str,
        nonce: &str,
        body: &str,
        signature: &str,
    ) -> Result<bool, VerificationError> {
        // 拼接签名数据
        let sign_str = format!("{}{}{}{}", timestamp, nonce, self.secret, body);

        // 计算 HMAC-SHA256
        let mut mac = HmacSha256::new_from_slice(self.secret.as_bytes())?;
        mac.update(sign_str.as_bytes());
        let expected = hex::encode(mac.finalize().into_bytes());

        // 安全比较
        Ok(hmac::compare_digest(expected.as_bytes(), signature.as_bytes()))
    }

    /// 验证企业微信签名
    pub fn verify_wecom(
        &self,
        msg_signature: &str,
        timestamp: &str,
        nonce: &str,
        echostr: &str,
    ) -> Result<bool, VerificationError> {
        // 排序参数
        let mut arr = vec![self.secret.clone(), timestamp.to_string(), nonce.to_string(), echostr.to_string()];
        arr.sort();

        // 计算 SHA1
        let sign_str = arr.join("");
        let mut hasher = Sha1::new();
        hasher.update(sign_str.as_bytes());
        let result = hasher.finalize();

        let expected = hex::encode(result);

        Ok(msg_signature == expected)
    }
}
```

```rust
            processed_messages: Arc::new(RwLock::new(HashSet::new())),
            message_ttl: ttl,
            cleanup_interval,
        };

        // 启动清理任务
        tokio::spawn(instance.clone().cleanup_task());

        instance
    }

    /// 检查是否为重放攻击
    pub async fn check(&self, message_id: &str, timestamp: i64) -> Result<bool, ReplayError> {
        // 1. 检查时间戳是否在有效范围内
        let now = chrono::Utc::now().timestamp();
        if (now - timestamp).abs() > self.message_ttl.as_secs() as i64 {
            return Err(ReplayError::TimestampExpired);

        // 2. 检查消息是否已处理
        let mut processed = self.processed_messages.write().await;

        if processed.contains(message_id) {
            return Err(ReplayError::DuplicateMessage);
        }

        // 3. 记录消息
        processed.insert(message_id.to_string());

        Ok(true)
    }

    async fn cleanup_task(self) {
        let mut interval = tokio::time::interval(self.cleanup_interval);

        loop {
            interval.tick().await;
```

```rust
// src/security/rate_limiter.rs

use std::collections::HashMap;

pub struct RateLimiter {
    // 用户限流记录
    user_limits: Arc<RwLock<HashMap<String, RateLimitRecord>>>,

    // IP 限流记录
    ip_limits: Arc<RwLock<HashMap<String, RateLimitRecord>>>,

    // 配置
    user_limit: usize,
    ip_limit: usize,
    window: Duration,
}

#[derive(Debug, Clone)]
struct RateLimitRecord {
    count: usize,
    window_start: Instant,
}

impl RateLimiter {
    pub fn new(user_limit: usize, ip_limit: usize, window: Duration) -> Self {
        Self {
            user_limits: Arc::new(RwLock::new(HashMap::new())),
            ip_limits: Arc::new(RwLock::new(HashMap::new())),
            user_limit,
            ip_limit,
            window,
        }
    }

    /// 检查用户限流
    pub async fn check_user(&self, user_id: &str) -> Result<bool, RateLimitError> {
        self.check(&self.user_limits, user_id, self.user_limit).await
    }

    /// 检查 IP 限流
    pub async fn check_ip(&self, ip: &str) -> Result<bool, RateLimitError> {
        self.check(&self.ip_limits, ip, self.ip_limit).await
    }

    async fn check(
        &self,
        storage: &Arc<RwLock<HashMap<String, RateLimitRecord>>>,
        key: &str,
        limit: usize,
    ) -> Result<bool, RateLimitError> {
        let mut limits = storage.write().await;
        let now = Instant::now();

        let record = limits.entry(key.to_string()).or_insert(RateLimitRecord {
            count: 0,
            window_start: now,
        });

        // 检查窗口是否过期
        if now.duration_since(record.window_start) > self.window {
            // 重置计数
            record.count = 1;
            record.window_start = now;
            return Ok(true);
        }

        // 检查是否超限
        if record.count >= limit {
            return Err(RateLimitError::LimitExceeded {
                limit,
                reset_after: self.window.saturating_sub(now.duration_since(record.window_start)),
            });
        }

        // 增加计数
        record.count += 1;

        Ok(true)
    }
}
```

```rust
// src/security/encryption.rs

use aes_gcm::{Aes256Gcm, Key, Nonce};
use aes_gcm::aead::{Aead, NewAead};
use keyring::Entry as KeyringEntry;
use rand::Rng;

pub struct ConfigEncryption {
    cipher: Aes256Gcm,
}

impl ConfigEncryption {
    /// 从系统密钥链获取或创建密钥
    pub fn new() -> Result<Self, EncryptionError> {
        let entry = KeyringEntry::new("desktop-agent", "encryption-key");

        // 尝试获取密钥
        let key_bytes = match entry.get_password() {
            Ok(key_str) => {
                // 从 Base64 解码
                base64::decode(key_str)?
            }
            Err(_) => {
                // 生成新密钥
                let key_bytes: [u8; 32] = rand::thread_rng().gen();
                let key_str = base64::encode(key_bytes);

                // 保存到密钥链
                entry.set_password(&key_str)?;

                key_bytes.to_vec()
            }
        };

        let key = Key::from_slice(&key_bytes);
        let cipher = Aes256Gcm::new(key);

        Ok(Self { cipher })
    }

    /// 加密数据
    pub fn encrypt(&self, plaintext: &str) -> Result<String, EncryptionError> {
        let nonce = Nonce::from_slice(&rand::thread_rng().gen::<[u8; 12]>());

        let ciphertext = self.cipher
            .encrypt(nonce, plaintext.as_bytes())
            .map_err(|_| EncryptionError::EncryptionFailed)?;

        // 组合 nonce 和 ciphertext
        let mut result = nonce.to_vec();
        result.extend(ciphertext);

        Ok(base64::encode(result))
    }

    /// 解密数据
    pub fn decrypt(&self, encrypted: &str) -> Result<String, EncryptionError> {
        let data = base64::decode(encrypted)?;

        if data.len() < 12 {
            return Err(EncryptionError::InvalidFormat);
        }

        let (nonce, ciphertext) = data.split_at(12);
        let nonce = Nonce::from_slice(nonce);

        let plaintext = self.cipher
            .decrypt(nonce, ciphertext)
            .map_err(|_| EncryptionError::DecryptionFailed)?;

        Ok(String::from_utf8(plaintext)?)
    }
}
```

```rust
// src/logging/sanitizer.rs

use regex::Regex;
use once_cell::sync::Lazy;

pub struct LogSanitizer;

impl LogSanitizer {
    /// 脱敏处理
    pub fn sanitize(text: &str) -> String {
        let mut result = text.to_string();

        // 脱敏敏感字段
        for (pattern, replacement) in SENSITIVE_PATTERNS.iter() {
            result = pattern.replace_all(&result, *replacement).to_string();
        }

        // 脱敏敏感信息
        for (pattern, replacement) in SENSITIVE_INFO_PATTERNS.iter() {
            result = pattern.replace_all(&result, *replacement).to_string();
        }

        result
    }
}

// 敏感字段模式
static SENSITIVE_PATTERNS: Lazy<Vec<(Regex, &str)>> = Lazy::new(|| {
    vec![
        (Regex::new(r"password['\"]?\s*[:=]\s*['\"]?[^'\"]+").unwrap(), "password=***"),
        (Regex::new(r"token['\"]?\s*[:=]\s*['\"]?[^'\"]{10,}").unwrap(), "token=***"),
        (Regex::new(r"secret['\"]?\s*[:=]\s*['\"]?[^'\"]{10,}").unwrap(), "secret=***"),
        (Regex::new(r"api_key['\"]?\s*[:=]\s*['\"]?[^'\"]{10,}").unwrap(), "api_key=***"),

        (Regex::new(r"private_key['\"]?\s*[:=]\s*['\"]?[^'\"]{20,}").unwrap(), "private_key=***"),
```

```rust
pub struct DataMinimizer;

impl DataMinimizer {
    /// 最小化会话数据
    pub fn minimize_session(session: &Session) -> Session {
        let mut minimized = session.clone();

        // 只保留必要的消息
        minimized.messages = minimized.messages
            .into_iter()
            .filter(|msg| {

                !self.contains_sensitive_info(&msg.content)
            })
            .collect();

        minimized
    }

    /// 最小化操作日志
    pub fn minimize_log(log: &OperationLog) -> OperationLog {
        let mut minimized = log.clone();

        // 移除操作数据中的敏感信息
        if let Ok(data) = serde_json::to_string_pretty(&log.operation_data) {
            minimized.operation_data = serde_json::from_str(
                &LogSanitizer::sanitize(&data)
            ).unwrap_or(log.operation_data.clone());
        }

        minimized
    }

    /// 检查是否包含敏感信息
    fn contains_sensitive_info(&self, text: &str) -> bool {
        let sanitized = LogSanitizer::sanitize(text);
        sanitized != text
    }

    /// 自动删除过期数据

        let mut result = CleanupResult::default();
```

```rust
impl AuditLogger {
    pub fn new(db: Arc<Database>, buffer_size: usize) -> Self {
        Self {
            db,
            buffer: Arc::new(Mutex::new(Vec::new())),
            buffer_size,
        }
    }

    /// 记录操作
    pub async fn log(&self, log: OperationLog) -> Result<(), AuditError> {
        let mut buffer = self.buffer.lock().await;

        buffer.push(log);

        // 缓冲区满了则批量写入
        if buffer.len() >= self.buffer_size {
            self.flush_inner(&mut buffer).await?;
        }

        Ok(())
    }

    /// 刷新缓冲区
    pub async fn flush(&self) -> Result<(), AuditError> {
        let mut buffer = self.buffer.lock().await;
        self.flush_inner(&mut buffer).await?;
        Ok(())
    }

    async fn flush_inner(&self, buffer: &mut Vec<OperationLog>) -> Result<(), AuditError> {
        if buffer.is_empty() {
            return Ok(());
        }

        self.db.batch_insert_operation_logs(buffer.clone()).await?;
        buffer.clear();

        Ok(())
    }
}
```
