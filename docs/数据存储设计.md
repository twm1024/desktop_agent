## 3. SQLite 数据库设计

### 3.1 数据库特性

- **数据库类型**: SQLite 3.x
- **文件位置**: `~/.desktop-agent/data/agent.db`
- **连接池大小**: 5
- **WAL 模式**: 启用（提升并发性能）
- **外键约束**: 启用

### 3.2 数据库表结构

#### 3.2.1 技能表 (skills)

```sql
CREATE TABLE skills (
    id TEXT PRIMARY KEY,                    -- 技能唯一标识
    name TEXT NOT NULL,                     -- 技能名称
    version TEXT NOT NULL,                  -- 版本号 (semver)
    description TEXT,                       -- 描述
    author TEXT,                            -- 作者
    tags TEXT,                              -- 标签 (JSON 数组)
    manifest TEXT NOT NULL,                 -- 技能清单 (JSON)
    enabled BOOLEAN DEFAULT 1,              -- 是否启用
    installed_at INTEGER NOT NULL,          -- 安装时间 (Unix timestamp)
    updated_at INTEGER NOT NULL,            -- 更新时间 (Unix timestamp)
    last_executed_at INTEGER,               -- 最后执行时间
    execution_count INTEGER DEFAULT 0,      -- 执行次数
    source TEXT,                            -- 来源 (official/community/local)
    checksum TEXT                           -- 文件校验和 (SHA256)
);

-- 索引
CREATE INDEX idx_skills_enabled ON skills(enabled);
CREATE INDEX idx_skills_name ON skills(name);
CREATE INDEX idx_skills_tags ON skills(tags);
```

**字段说明**:

| 字段 | 类型 | 说明 |
|------|------|------|
| id | TEXT | 技能唯一标识，格式: `{author}.{name}` |
| name | TEXT | 技能显示名称 |
| version | TEXT | 遵循语义化版本号 (Semantic Versioning) |
| tags | TEXT | JSON 数组，如 `["文件", "批量处理"]` |
| manifest | TEXT | 完整的技能清单配置（JSON 格式） |
| enabled | BOOLEAN | 1 表示启用，0 表示禁用 |
| checksum | TEXT | 技能文件 SHA256 校验和，用于完整性验证 |

#### 3.2.2 会话表 (sessions)

```sql
CREATE TABLE sessions (
    id TEXT PRIMARY KEY,                    -- 会话 ID
    user_id TEXT NOT NULL,                  -- 用户 ID
    chat_id TEXT NOT NULL,                  -- 聊天 ID
    platform TEXT NOT NULL,                 -- 平台 (feishu/wecom/dingtalk)
    current_intent TEXT,                    -- 当前意图
    slots TEXT,                             -- 槽位数据 (JSON)
    messages TEXT,                          -- 消息历史 (JSON 数组)
    created_at INTEGER NOT NULL,            -- 创建时间 (Unix timestamp)
    last_active INTEGER NOT NULL,           -- 最后活跃时间 (Unix timestamp)
    state TEXT DEFAULT 'active',            -- 状态 (active/completed/abandoned)
    metadata TEXT                           -- 元数据 (JSON)
);

-- 索引
CREATE INDEX idx_sessions_user ON sessions(user_id);
CREATE INDEX idx_sessions_chat ON sessions(chat_id);
CREATE INDEX idx_sessions_platform ON sessions(platform);
CREATE INDEX idx_sessions_last_active ON sessions(last_active);
CREATE INDEX idx_sessions_state ON sessions(state);
```

**字段说明**:

| 字段 | 类型 | 说明 |
|------|------|------|
| id | TEXT | 格式: `{platform}:{user_id}:{chat_id}` |
| platform | TEXT | 平台标识，枚举值 |
| state | TEXT | 会话状态: active（活跃）、completed（完成）、abandoned（放弃） |
| slots | TEXT | JSON 对象，如 `{"file_path": "/home/user/file.txt", "pattern": "*.jpg"}` |
| messages | TEXT | JSON 数组，包含用户和机器人的所有消息 |

**slots 示例**:
```json
{
  "file_path": "/home/user/pictures",
  "pattern": "IMG_*",
  "template": "Photo_{:03d}.jpg"
}
```

**messages 示例**:
```json
[
  {
    "role": "user",
    "content": "帮我重命名桌面上的图片",
    "timestamp": 1703222400,
    "message_type": "text"
  },
  {
    "role": "bot",
    "content": "好的，请告诉我图片文件的路径",
    "timestamp": 1703222405,
    "message_type": "text"
  }
]
```

#### 3.2.3 操作日志表 (operation_logs)

```sql
CREATE TABLE operation_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp INTEGER NOT NULL,             -- 时间戳 (Unix timestamp)
    user_id TEXT NOT NULL,                  -- 用户 ID
    platform TEXT NOT NULL,                 -- 平台
    operation_type TEXT NOT NULL,           -- 操作类型
    operation_data TEXT NOT NULL,           -- 操作数据 (JSON)
    result TEXT NOT NULL,                   -- 结果 (JSON)
    skill_id TEXT,                          -- 关联的技能 ID
    session_id TEXT,                        -- 关联的会话 ID
    duration_ms INTEGER,                    -- 执行耗时 (毫秒)
    status TEXT NOT NULL,                   -- 状态 (success/failure/partial)
    error_message TEXT,                     -- 错误信息
    ip_address TEXT,                        -- IP 地址
    user_agent TEXT                         -- User Agent
);

-- 索引
CREATE INDEX idx_logs_user ON operation_logs(user_id);
CREATE INDEX idx_logs_platform ON operation_logs(platform);
CREATE INDEX idx_logs_timestamp ON operation_logs(timestamp);
CREATE INDEX idx_logs_type ON operation_logs(operation_type);
CREATE INDEX idx_logs_skill ON operation_logs(skill_id);
CREATE INDEX idx_logs_session ON operation_logs(session_id);
CREATE INDEX idx_logs_status ON operation_logs(status);
```

**字段说明**:

| 字段 | 类型 | 说明 |
|------|------|------|
| operation_type | TEXT | 操作类型，如 `file:delete`、`skill:execute` |
| operation_data | TEXT | JSON 格式的操作参数 |
| result | TEXT | JSON 格式的执行结果 |
| status | TEXT | success（成功）、failure（失败）、partial（部分成功） |

**operation_data 示例**:
```json
{
  "action": "file:delete",
  "path": "/home/user/test.txt",
  "recursive": false
}
```

**result 示例**:
```json
{
  "success": true,
  "message": "文件删除成功",
  "data": {
    "deleted_files": ["/home/user/test.txt"],
    "count": 1
  }
}
```

```sql
CREATE TABLE users (
```

**字段说明**:

### 4.3 权限配置 (config/permissions.yaml)

```yaml
# 全局权限策略
policy:
  default_deny: true           # 默认拒绝未声明的操作
  require_confirmation: true   # 敏感操作需要确认

# 权限定义
permissions:
  # 文件权限

    default_access: "read"     # 默认访问权限: none/read/readwrite
    allowed_paths:
      - "~/documents"
      - "~/downloads"
      - "~/desktop"
      - "~/pictures"
    denied_paths:
      - "/etc"
      - "/usr"
      - "/bin"
      - "/sbin"
      - "C:\\Windows"
      - "C:\\Program Files"

  # 网络权限
  network:
    default_allow: false       # 默认禁止网络访问
    allowed_domains: []
    denied_domains:
      - "*.malicious.com"
    max_requests_per_minute: 100

  # 系统权限
  system:
    allow_execute_command: false
    allow_process_control: false
    allow_system_info: true
    allow_screenshot: true
    allow_clipboard: true

# 角色权限
roles:
  admin:
    name: "超级管理员"
    permissions:
      - "*:*:*"                # 所有权限

  user:
    name: "普通用户"
    permissions:
      - "file:read:~/documents"
      - "file:read:~/downloads"
      - "file:write:~/downloads"
      - "file:read:~/desktop"
      - "system:info:read"
      - "system:screenshot"
      - "system:clipboard"

  readonly:
    name: "只读用户"
    permissions:
      - "file:read:~/documents"
      - "file:read:~/downloads"
      - "file:read:~/desktop"
      - "system:info:read"

# 敏感操作配置
sensitive_operations:
  operations:
    - type: "file:delete"
      requires_confirmation: true
      confirmation_message: "确定要删除文件 {path} 吗？"

    - type: "file:delete:recursive"

    - type: "system:execute"
      requires_confirmation: true
      confirmation_message: "确定要执行命令 {command} 吗？"
```

### 4.5 技能清单示例 (skills/file-rename/skill.yaml)

```yaml
# 技能清单
name: "批量文件重命名"
version: "1.2.0"
description: "批量重命名文件，支持多种命名模板"
author: "Desktop Agent Team"
tags: ["文件", "批量处理", "重命名"]

# 主入口文件
main: "index.py"

# 依赖
dependencies:
  python: ">=3.8"
  pip:
    - "pathlib"

# 权限声明
permissions:
  file:
    - path: "~/documents"
      access: "read"
    - path: "~/downloads"
      access: "readwrite"
  network: []

# 对话配置
dialog:
  # 意图定义
  intents:
    - name: "batch_rename"
      examples:
        - "批量重命名文件"
        - "帮我重命名 {path} 下的文件"
        - "把 {path} 的文件按 {pattern} 重命名"
        - "批量修改文件名"
      triggers:
        - keyword: "重命名"
        - regex: "(批量)?(重命名|改名为|改成).*文件"
        - nlu: "file.batch_rename"

  # 槽位定义
  slots:
    - name: "path"
      type: "string"
      required: true
      description: "要处理的目录路径"
      prompts:
        - "请告诉我要处理哪个目录？"
        - "文件在哪个路径下？"

    - name: "pattern"
      type: "string"
      required: false
      default: "*"
      description: "文件匹配模式"
      prompts:
        - "需要匹配哪些文件？（如 *.jpg）"

    - name: "template"
```

```rust
// src/database/transaction.rs

pub struct TransactionManager {
    pool: SqlitePool,
}

impl TransactionManager {
    pub async fn begin_transaction(&self) -> Result<sqlx::Transaction<sqlx::Sqlite>, DatabaseError> {
        Ok(self.pool.begin().await?)
    }

    pub async fn execute_in_transaction<F, R>(&self, f: F) -> Result<R, DatabaseError>
    where
        F: FnOnce(&mut sqlx::Transaction<sqlx::Sqlite>) -> futures::future::BoxFuture<'_, Result<R, DatabaseError>>,
    {
        let mut tx = self.begin_transaction().await?;

        let result = f(&mut tx).await;

        match result {
            Ok(r) => {
                tx.commit().await?;
                Ok(r)
            }
            Err(e) => {
                tx.rollback().await?;
                Err(e)
            }
        }
    }
}
```

## 7. 数据备份与恢复

### 7.1 备份策略

```rust
// src/backup/backup_manager.rs

pub struct BackupManager {
    config_dir: PathBuf,
    data_dir: PathBuf,
    backup_dir: PathBuf,
}

impl BackupManager {
    /// 创建备份
    pub async fn create_backup(&self) -> Result<BackupInfo, BackupError> {
        let timestamp = chrono::Local::now().format("%Y%m%d_%H%M%S").to_string();
        let backup_id = format!("backup_{}", timestamp);
        let backup_path = self.backup_dir.join(&backup_id);

        // 创建备份目录
        fs::create_dir_all(&backup_path)?;

        // 1. 备份配置文件
        self.backup_config(&backup_path).await?;

        // 2. 备份数据库
        self.backup_database(&backup_path).await?;

        // 3. 备份技能清单
        self.backup_skill_manifests(&backup_path).await?;

        // 4. 清理旧备份
        self.cleanup_old_backups().await?;

        Ok(BackupInfo {
            id: backup_id,
            path: backup_path,
            timestamp: chrono::Local::now(),
            size: self.calculate_size(&backup_path),
        })
    }

    /// 恢复备份
    pub async fn restore_backup(&self, backup_id: &str) -> Result<(), BackupError> {
        let backup_path = self.backup_dir.join(backup_id);

        if !backup_path.exists() {
            return Err(BackupError::NotFound(backup_id.to_string()));
        }

        // 1. 停止服务
        self.stop_services().await?;

        // 2. 创建当前状态备份
        self.create_backup().await?;

        // 3. 恢复配置文件
        self.restore_config(&backup_path).await?;

        // 4. 恢复数据库
        self.restore_database(&backup_path).await?;

        // 5. 重启服务
        self.restart_services().await?;

        Ok(())
    }
}
```

```rust
pub struct MigrationRunner {
    pool: SqlitePool,
    migrations_dir: PathBuf,
}

impl MigrationRunner {
    pub async fn run_migrations(&self) -> Result<(), MigrationError> {
        // 获取已应用的迁移
        let applied = self.get_applied_migrations().await?;

        // 获取待应用的迁移
        let pending = self.get_pending_migrations(&applied)?;

        // 按顺序应用迁移
        for migration in pending {
            self.apply_migration(&migration).await?;
        }

        Ok(())
    }

    async fn apply_migration(&self, migration: &Migration) -> Result<(), MigrationError> {
        let mut tx = self.pool.begin().await?;

        // 执行迁移 SQL
        sqlx::query(&migration.sql)
            .execute(&mut *tx)
            .await?;

        // 记录迁移
        sqlx::query!(
            "INSERT INTO _migrations (name, applied_at) VALUES (?, ?)",
            migration.name,
            chrono::Utc::now().timestamp()
        )
        .execute(&mut *tx)
        .await?;

        tx.commit().await?;

        Ok(())
    }
}
```

```rust
// src/security/encryption.rs

pub struct ConfigEncryption {
    cipher: Aes256Gcm,
}

impl ConfigEncryption {
    /// 从系统密钥链获取密钥
    pub fn new() -> Result<Self, EncryptionError> {
        let key = self::get_or_create_key()?;
        let cipher = Aes256Gcm::new_from_slice(&key)?;
        Ok(Self { cipher })
    }

    /// 加密配置
    pub fn encrypt(&self, plaintext: &str) -> Result<String, EncryptionError> {
        let nonce = Aes256Gcm::generate_nonce(&mut OsRng);
        let ciphertext = self.cipher.encrypt(&nonce, plaintext.as_bytes())?;

        let mut result = nonce.to_vec();
        result.extend(ciphertext);

        Ok(base64::encode(result))
    }

    /// 解密配置
    pub fn decrypt(&self, ciphertext: &str) -> Result<String, EncryptionError> {
        let data = base64::decode(ciphertext)?;
        let (nonce, ciphertext) = data.split_at(12);

        let plaintext = self.cipher.decrypt(nonce.into(), ciphertext)?;
        Ok(String::from_utf8(plaintext)?)
    }
}
```

```rust
// src/logging/sanitizer.rs

pub struct LogSanitizer {
    patterns: Vec<(Regex, String)>,
}

impl LogSanitizer {
    pub fn new() -> Self {
        Self {
            patterns: vec![
                (Regex::new(r"password['\"]?\s*[:=]\s*['\"]?[^'\"]+").unwrap(), "password=***"),
                (Regex::new(r"token['\"]?\s*[:=]\s*['\"]?[^'\"]{10,}").unwrap(), "token=***"),
                (Regex::new(r"secret['\"]?\s*[:=]\s*['\"]?[^'\"]{10,}").unwrap(), "secret=***"),
                (Regex::new(r"api_key['\"]?\s*[:=]\s*['\"]?[^'\"]{10,}").unwrap(), "api_key=***"),
            ],
        }
    }

    pub fn sanitize(&self, text: &str) -> String {
        let mut result = text.to_string();
        for (pattern, replacement) in &self.patterns {
            result = pattern.replace_all(&result, *replacement).to_string();
        }
        result
    }
}
```
